Passo 3

Configurar o Workflow de Deploy

No GitHub, crie o arquivo .github/workflows/main.yml. Este script vai copiar os arquivos e pedir para o PM2 reiniciar a API no servidor:

name: Deploy Back-end
on: [push]
jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Enviar Arquivos (SCP)
        uses: appleboy/scp-action@v0.1.7
        with:
          host: ${{ secrets.SSH_HOST }}
          username: root
          key: ${{ secrets.SSH_KEY }}
          source: "./*"
          target: "/var/inetpub/wwwroot/projetos/${{ secrets.ALUNO_RA }}/back/${{ secrets.ALUNO_PROJETO }}"

      - name: Reiniciar API via SSH
        uses: appleboy/ssh-action@master
        with:
          host: ${{ secrets.SSH_HOST }}
          username: root
          key: ${{ secrets.SSH_KEY }}
          script: |
            cd /var/inetpub/wwwroot/projetos/${{ secrets.ALUNO_RA }}/back/${{ secrets.ALUNO_PROJETO }}
            npm install --production
            pm2 delete "api-${{ secrets.ALUNO_RA }}" || true
            pm2 start server.js --name "api-${{ secrets.ALUNO_RA }}"
